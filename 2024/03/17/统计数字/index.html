n=int(input())
dir={}
for i in range(n):
    num=int(input())
    if num in dir:
        dir[num]+=1
    else:
        dir[num]=1
dir=sorted(dir.items(),key=lambda x:x[0])
for i in dir:
    print(i[0],i[1])
