```python
import turtle

# 使用Python的turtle库绘制简单的山水画
def draw_mountain(x, y):
    turtle.penup()
    turtle.goto(x, y)
    turtle.pendown()
    turtle.begin_fill()
    turtle.circle(50, 180)
    turtle.right(90)
    turtle.forward(100)
    turtle.right(90)
    turtle.circle(50, 180)
    turtle.end_fill()

def draw_scenery():
    turtle.speed(1)
    draw_mountain(-150, 150)
    draw_mountain(150, 150)
    turtle.done()

draw_scenery()
```
